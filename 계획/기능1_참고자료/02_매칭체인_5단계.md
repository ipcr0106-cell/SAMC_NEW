## 1. 매칭 단계 정의

| 순서 | 단계 | 조회 대상 컬럼 | 신뢰도 | 조건 |
|------|------|---------------|--------|------|
| 1 | **exact_name** | `name_ko` | 1.0 | 입력 이름 trim 후 정확 일치 |
| 2 | **ins_number** | `ins_number` | 1.0 | 입력에 INS 번호 존재 |
| 3 | **cas_number** | `cas_number` | 1.0 | 입력에 CAS 번호 존재 |
| 4 | **scientific_name** | `name_scientific` | 1.0 | 입력 이름이 학명인 경우 (정규식 판정) |
| 5 | **fuzzy** | `search_ingredients_trgm(q,k)` RPC | 0.7 | 앞 4단계 실패 시 trgm 유사도 |
| ★ | **llm_normalize** (선택) | Claude API | 0.5 | 퍼지도 실패 시 이름 정규화 후 재시도 (개발계획서 §4-1 4차) |

> 1~5는 newsamc 원본, ★은 개발계획서 v6의 "4차: Claude API로 이름 정규화 후 재매칭" 추가.

## 2. 판정 결과 타입 (Pydantic)

```python
# backend/models/judgment.py  (신규 제안)
from typing import Literal, Optional
from pydantic import BaseModel

IngredientVerdict = Literal['permitted', 'restricted', 'prohibited', 'unidentified']
MatchMethod = Literal['exact_name', 'ins_number', 'cas_number',
                      'scientific_name', 'fuzzy', 'llm_normalize']


class Ingredient(BaseModel):
    name: str
    name_original: Optional[str] = None
    percentage: Optional[float] = None
    ins: Optional[str] = None
    cas: Optional[str] = None
    chemical_name: Optional[str] = None
    part: Optional[str] = None
    is_allergen: Optional[bool] = None
    sub_ingredients: Optional[list['Ingredient']] = None


class IngredientMatchResult(BaseModel):
    ingredient: Ingredient
    verdict: IngredientVerdict
    match_method: Optional[MatchMethod]
    matched_db_id: Optional[str]
    confidence: float                       # 0.0 ~ 1.0
    conditions: Optional[str] = None        # restricted 일 때 조건 텍스트
    matched_name_ko: Optional[str] = None   # DB에서 매칭된 공식 한국어명


class AggregationResult(BaseModel):
    total: int
    permitted: int
    restricted: int
    prohibited: int
    unidentified: int
    results: list[IngredientMatchResult]
    escalations: list[dict]   # 아래 §4 참고
```

## 3. 5단계 매칭 함수 (초안)

```python
# backend/services/step1_ingredients_check.py
from asyncpg import Connection
from backend.models.judgment import (
    Ingredient, IngredientMatchResult, AggregationResult, IngredientVerdict
)

EXACT_MATCH_CONFIDENCE = 1.0
FUZZY_MATCH_CONFIDENCE = 0.7
LLM_NORMALIZE_CONFIDENCE = 0.5

_VERDICT_MAP: dict[str, IngredientVerdict] = {
    "permitted": "permitted",
    "restricted": "restricted",
    "prohibited": "prohibited",
}


async def match_ingredient(db: Connection, ing: Ingredient) -> IngredientMatchResult:
    """기능1 5단계 원재료 매칭 체인."""
    name = (ing.name or "").strip()
    if not name:
        return _unidentified(ing)

    # ── 1단계: 정확 이름 매칭 ──────────────────────────
    row = await db.fetchrow("""
        SELECT id, name_ko, allowed_status, usage_condition, law_source
          FROM ingredient_list
         WHERE name_ko = $1
         LIMIT 1
    """, name)
    if row:
        return _build_result(ing, row, "exact_name", EXACT_MATCH_CONFIDENCE)

    # ── 2단계: INS 번호 매칭 ───────────────────────────
    if ing.ins:
        row = await db.fetchrow("""
            SELECT id, name_ko, allowed_status, usage_condition, law_source
              FROM ingredient_list
             WHERE ins_number = $1
             LIMIT 1
        """, ing.ins)
        if row:
            return _build_result(ing, row, "ins_number", EXACT_MATCH_CONFIDENCE)

    # ── 3단계: CAS 번호 매칭 ───────────────────────────
    if ing.cas:
        row = await db.fetchrow("""
            SELECT id, name_ko, allowed_status, usage_condition, law_source
              FROM ingredient_list
             WHERE cas_number = $1
             LIMIT 1
        """, ing.cas)
        if row:
            return _build_result(ing, row, "cas_number", EXACT_MATCH_CONFIDENCE)

    # ── 4단계: 학명 매칭 (입력 name이 학명일 수도) ─────
    row = await db.fetchrow("""
        SELECT id, name_ko, allowed_status, usage_condition, law_source
          FROM ingredient_list
         WHERE name_scientific ILIKE $1
         LIMIT 1
    """, f"%{name}%")
    if row:
        return _build_result(ing, row, "scientific_name", EXACT_MATCH_CONFIDENCE)

    # ── 5단계: 퍼지(trgm) 매칭 ─────────────────────────
    row = await db.fetchrow("""
        SELECT id, name_ko, allowed_status, usage_condition, similarity
          FROM search_ingredients_trgm($1, 1)
    """, name)
    if row and row["similarity"] >= 0.3:     # 임계값 (튜닝 필요)
        return _build_result(ing, row, "fuzzy", FUZZY_MATCH_CONFIDENCE)

    # ── ★단계: Claude API 정규화 재시도 (선택) ─────────
    # normalized = await normalize_ingredient_name_via_claude(name)
    # if normalized != name:
    #     return await match_ingredient(db, ing.copy(update={"name": normalized}))

    return _unidentified(ing)


def _build_result(ing: Ingredient, row, method: str, confidence: float) -> IngredientMatchResult:
    return IngredientMatchResult(
        ingredient=ing,
        verdict=_VERDICT_MAP.get(row["allowed_status"], "unidentified"),
        match_method=method,
        matched_db_id=str(row["id"]),
        confidence=confidence,
        conditions=row.get("usage_condition"),
        matched_name_ko=row["name_ko"],
    )


def _unidentified(ing: Ingredient) -> IngredientMatchResult:
    return IngredientMatchResult(
        ingredient=ing, verdict="unidentified",
        match_method=None, matched_db_id=None, confidence=0.0,
    )
```

## 4. 집계 + 에스컬레이션

```python
async def run_ingredient_match_chain(
    db: Connection, ingredients: list[Ingredient]
) -> AggregationResult:
    results: list[IngredientMatchResult] = []
    escalations: list[dict] = []

    for ing in ingredients:
        r = await match_ingredient(db, ing)
        results.append(r)

        if r.verdict == "prohibited":
            escalations.append({
                "module_id": "F1",
                "trigger_type": "prohibited_detected",
                "confidence_score": r.confidence,
                "reason": f'금지 원료 감지: "{ing.name}"',
            })
        elif r.verdict == "unidentified":
            escalations.append({
                "module_id": "F1",
                "trigger_type": "low_confidence",
                "confidence_score": 0.0,
                "reason": f'미확인 원료: "{ing.name}" — DB 매칭 실패',
            })

    return AggregationResult(
        total=len(results),
        permitted=sum(1 for r in results if r.verdict == "permitted"),
        restricted=sum(1 for r in results if r.verdict == "restricted"),
        prohibited=sum(1 for r in results if r.verdict == "prohibited"),
        unidentified=sum(1 for r in results if r.verdict == "unidentified"),
        results=results,
        escalations=escalations,
    )
```

## 5. Step 0 — forbidden_ingredients 게이트

개발계획서 §4-1 "수입불가 판정 시 파이프라인 즉시 중단"을 반영:

```python
async def check_forbidden_first(
    db: Connection, ingredients: list[Ingredient]
) -> Optional[dict]:
    """절대 금지 원재료가 하나라도 있으면 즉시 수입불가 반환."""
    names = [ing.name for ing in ingredients]
    rows = await db.fetch("""
        SELECT name_ko, category, law_source, reason
          FROM forbidden_ingredients
         WHERE name_ko = ANY($1)
            OR EXISTS (
               SELECT 1 FROM unnest(aliases) a
                WHERE a = ANY($1)
            )
    """, names)
    if rows:
        return {
            "import_possible": False,
            "verdict": "수입불가 — 절대 금지 원료 포함",
            "forbidden_hits": [dict(r) for r in rows],
        }
    return None
```

## 8. 성능·캐시 관련 힌트 (newsamc 설계 13번에서 참고)

- trgm 인덱스는 한 번 만들면 조회 수 ms 단위 (12,000건 원료에서 확인)
- `ingredient_list` 전체 메모리 캐싱은 **비권장** (개정 빈번). DB 조회가 충분히 빠름.
- 연속 동일 원료명 재조회가 많을 때만 `functools.lru_cache(maxsize=1024)` 데코레이터로 per-request 캐시.
