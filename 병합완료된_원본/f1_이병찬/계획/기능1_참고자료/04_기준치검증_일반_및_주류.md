**출력 스키마** (Pydantic)

```python
LimitCheckStatus = Literal['pass', 'fail', 'warning', 'no_data']
LimitCategory = Literal['additive', 'heavy_metal', 'microbe',
                        'pesticide', 'alcohol']

class LimitCheckResult(BaseModel):
    item_name: str
    category: LimitCategory
    max_limit: str              # "0.6 g/kg" or "불검출" (TEXT)
    actual_value: Optional[str] = None
    status: LimitCheckStatus
    regulation_ref: Optional[str] = None

class StandardsCheckResult(BaseModel):
    food_type: str
    overall_status: Literal['pass', 'fail', 'review_needed']
    checks: list[LimitCheckResult]
    violations: list[LimitCheckResult]
    compound_results: list[dict]    # 복합 합산 결과
    escalations: list[dict]
```

### 2-2. 개별 판정

```python
from backend.utils.unit_converter import convert_unit   # 05 문서 참조

async def check_single_limit(
    ingredient: Ingredient, threshold_row: dict,
    process_conditions: dict
) -> LimitCheckResult:
    # 조건 체크 (예: 가열제품에 한함)
    cond = threshold_row.get("condition_text") or ""
    if "가열" in cond and not process_conditions.get("is_heated"):
        return LimitCheckResult(
            item_name=ingredient.name,
            category=threshold_row.get("standard_type") or "additive",
            max_limit=f"{threshold_row['threshold_value']} {threshold_row['unit']}",
            status="no_data",
            regulation_ref=(f"{threshold_row['law_source']} "
                            f"{threshold_row.get('law_article') or ''}".strip()),
        )

    if ingredient.percentage is None:
        # 실측값 미제공
        return LimitCheckResult(
            item_name=ingredient.name,
            category=threshold_row.get("standard_type") or "additive",
            max_limit=str(threshold_row.get("threshold_text")
                          or f"{threshold_row['threshold_value']} {threshold_row['unit']}"),
            status="no_data",
            regulation_ref=f"{threshold_row['law_source']} {threshold_row.get('law_article','')}",
        )

    # 단위 환산 (conversion_factor 적용)
    actual = convert_unit(
        ingredient.percentage, source_unit="%",       # 가정: 입력이 % 기준
        target_unit=threshold_row["unit"],
        factor=threshold_row.get("conversion_factor"),
    )

    limit_val = float(threshold_row["threshold_value"])
    status: LimitCheckStatus = "fail" if actual > limit_val else "pass"

    return LimitCheckResult(
        item_name=ingredient.name,
        category=threshold_row.get("standard_type") or "additive",
        max_limit=f"{limit_val} {threshold_row['unit']}",
        actual_value=f"{actual} {threshold_row['unit']}",
        status=status,
        regulation_ref=f"{threshold_row['law_source']} {threshold_row.get('law_article','')}",
    )
```

### 2-3. 복합 합산 (타르색소 등)

```python
async def evaluate_compound_groups(
    db: Connection, food_type: str, per_item_results: list[LimitCheckResult],
    ingredients: list[Ingredient],
) -> list[dict]:
    """compound_group이 있는 성분들의 합산값이 그룹 기준치를 초과하는지."""
    groups: dict[str, dict] = {}
    for r in per_item_results:
        if r.status not in ("pass", "fail"):
            continue
        row = await db.fetchrow("""
            SELECT compound_group, is_compound_limit FROM thresholds
             WHERE ingredient_name = $1 AND food_type = $2 AND is_verified = true
             LIMIT 1
        """, r.item_name, food_type)
        if not row or not row["compound_group"]:
            continue
        grp = row["compound_group"]
        groups.setdefault(grp, {"sum": 0.0, "members": []})
        # actual_value 파싱
        if r.actual_value:
            try:
                groups[grp]["sum"] += float(r.actual_value.split()[0])
            except ValueError:
                pass
        groups[grp]["members"].append(r.item_name)

    results = []
    for grp_name, data in groups.items():
        limit_row = await db.fetchrow("""
            SELECT threshold_value, unit, law_article, law_source
              FROM thresholds
             WHERE compound_group = $1 AND is_compound_limit = true
               AND food_type = $2 AND is_verified = true
             LIMIT 1
        """, grp_name, food_type)
        if not limit_row:
            continue
        total = data["sum"]
        limit = float(limit_row["threshold_value"])
        results.append({
            "group": grp_name,
            "members": data["members"],
            "total": total,
            "limit": limit,
            "unit": limit_row["unit"],
            "status": "fail" if total > limit else "pass",
            "law_ref": f"{limit_row['law_source']} {limit_row['law_article']}",
        })
    return results
```

## 3. 주류 안전기준 검증

### 3-1. 알코올 도수 경계치

newsamc `LIQUOR_BOUNDARY_THRESHOLDS`:

```python
LIQUOR_BOUNDARY_THRESHOLDS = {
    "non_alcohol_max": 0.5,   # 미만: 비알코올
    "boundary_max":    1.0,   # [0.5, 1.0): 경계치 → 에스컬레이션
    # 1.0 이상: 확실한 주류 → 주류 기준 검증
}

def is_alcohol_boundary(percentage: Optional[float]) -> bool:
    if percentage is None:
        return False
    return (LIQUOR_BOUNDARY_THRESHOLDS["non_alcohol_max"]
            <= percentage
            < LIQUOR_BOUNDARY_THRESHOLDS["boundary_max"])
```

### 3-2. 주류 4대 검사 항목

```python
LIQUOR_CHECK_ITEMS = [
    {"name": "메탄올",        "standard_type": "contaminant"},
    {"name": "알데히드",      "standard_type": "contaminant"},
    {"name": "퓨젤유",        "standard_type": "contaminant"},
    {"name": "에탄올(주정도)", "standard_type": "contaminant"},
]

async def check_liquor_safety(
    db: Connection, food_type_code: str,
    alcohol_percentage: Optional[float] = None,
) -> dict:
    escalations: list[dict] = []
    checks: list[LimitCheckResult] = []

    # 4개 항목 병렬 조회
    tasks = []
    for item in LIQUOR_CHECK_ITEMS:
        tasks.append(db.fetchrow("""
            SELECT threshold_value, threshold_text, unit, law_source, law_article
              FROM thresholds
             WHERE standard_type = $1
               AND ingredient_name ILIKE '%' || $2 || '%'
               AND is_verified = true
             LIMIT 1
        """, item["standard_type"], item["name"]))
    rows = await asyncio.gather(*tasks)

    for item, row in zip(LIQUOR_CHECK_ITEMS, rows):
        if row:
            limit_str = (row["threshold_text"]
                         or f"{row['threshold_value']} {row['unit']}")
            ref = f"{row['law_source']} {row.get('law_article','')}".strip()
        else:
            limit_str = "기준 미등록"
            ref = None
        checks.append(LimitCheckResult(
            item_name=item["name"], category="alcohol",
            max_limit=limit_str, status="no_data",
            regulation_ref=ref,
        ))

    # 경계치 에스컬레이션
    if is_alcohol_boundary(alcohol_percentage):
        escalations.append({
            "module_id": "F1",
            "trigger_type": "alcohol_boundary",
            "reason": (f"알코올 함량 {alcohol_percentage}% — "
                       f"주류/비주류 경계치. 수동 확인 필요"),
        })

    violations = [c for c in checks if c.status == "fail"]
    return {
        "food_type_code": food_type_code,
        "alcohol_percentage": alcohol_percentage,
        "overall_status": "fail" if violations else "review_needed",
        "checks": [c.model_dump() for c in checks],
        "violations": [v.model_dump() for v in violations],
        "escalations": escalations,
    }
```

### 3-3. 주류 분기 조건

기능2(아람)에서 넘어오는 `food_type` 또는 `is_alcohol` 플래그로 분기:

```python
async def run_standards_check(
    db, ingredients, food_type, food_type_result, process_conditions,
) -> StandardsCheckResult:
    is_alcohol = food_type_result.get("is_alcohol") or \
                 food_type in ("증류주", "발효주", "기타주류", "주류")
    alcohol_pct = process_conditions.get("alcohol_percentage")

    # 공통 일반식품 기준치 검증
    general = await check_general_limits(db, food_type, ingredients, process_conditions)

    if is_alcohol or is_alcohol_boundary(alcohol_pct):
        liquor = await check_liquor_safety(db, food_type, alcohol_pct)
        # 두 결과 병합
        general["checks"].extend(liquor["checks"])
        general["violations"].extend(liquor["violations"])
        general["escalations"].extend(liquor["escalations"])

    return general
```

## 4. 전체 판정 규칙표

| 조건 | `overall_status` | 최종 `import_possible` |
|------|-----------------|----------------------|
| 어떤 check라도 `status="fail"` | `fail` | `false` |
| 어떤 compound_result라도 `status="fail"` | `fail` | `false` |
| 모든 check가 `no_data` | `review_needed` | 담당자 판정 |
| 알코올 경계치 (`is_alcohol_boundary=true`) | `review_needed` | 담당자 판정 |
| 나머지 | `pass` | `true` |

## 5. 단위 테스트 시나리오

| 케이스 | 입력 | 기대 |

| 안식향산나트륨 0.5 < 0.6 | `{name:"안식향산나트륨", percentage:0.05}`, food_type="과채음료" | `pass` |
| 안식향산나트륨 0.8 > 0.6 | `{percentage:0.08}` | `fail` |
| 조건 미충족 | 기준 "가열제품에 한함" + `is_heated=false` | `no_data` |
| 타르색소 합산 초과 | 개별은 pass인데 합산 초과 | compound_results fail |
| 주류 + 경계치 | `alcohol_percentage=0.7` | `review_needed` + 에스컬레이션 |
| `is_verified=false` 레코드 | | 쿼리에서 제외됨 |
