### 권한 로직 (팀컨벤션 + 개발계획서 §3-5)

```python
@router.delete("/admin/db/{table}/{id}")
async def delete_row(table: str, id: str, user = Depends(get_current_user)):
    if table not in ("ingredient_list","thresholds","forbidden_ingredients"):
        raise HTTPException(404, detail={"error":"UNKNOWN_TABLE"})
    row = await db.fetchrow(f"SELECT created_by FROM {table} WHERE id=$1", id)
    if not row:
        raise HTTPException(404, detail={"error":"NOT_FOUND"})
    if row["created_by"] is None or row["created_by"] != user.id:
        raise HTTPException(403, detail={
            "error":"FORBIDDEN",
            "message":"본인이 추가한 항목만 삭제할 수 있습니다.",
            "feature": 1,
        })
    await db.execute(f"DELETE FROM {table} WHERE id=$1", id)
    return {"ok": True}
```

## 11. 통합 엔트리 (SSE 파이프라인 연결점)

공통 인프라 담당자와 합의해야 할 부분. 기능1 서비스 레이어가 반환하는 결과를 `pipeline_steps.ai_result` JSONB에 저장.

```python
# 예시 (공통 인프라 담당 파일에 포함될 영역)
async def execute_feature1(case_id: str):
    data = await load_case_inputs(case_id)
    result = await run_feature1(
        db, ingredients=data.ingredients,
        food_type=data.food_type_from_feature2,
        process_conditions=data.process_conditions,
    )
    await db.execute("""
        INSERT INTO pipeline_steps (case_id, step_key, step_name, status, ai_result)
        VALUES ($1, '1', 'import_check', 'waiting_review', $2)
        ON CONFLICT (case_id, step_key) DO UPDATE
          SET ai_result=$2, status='waiting_review', updated_at=now()
    """, case_id, json.dumps(result.model_dump(), ensure_ascii=False))
    return result
```

> 병찬은 `run_feature1`만 책임. `execute_feature1` 래퍼는 인프라 담당 영역.
